6. Reverse Integer

class Solution {
    public int reverse(int x) {
        int res = 0;
        while (x != 0) {
            res *= 10;
            res += x % 10;
            x /= 10;
            if (Integer.MAX_VALUE / 10 < Math.abs(res)) {
                return 0;
            }
        }
        return res;
    }
}
